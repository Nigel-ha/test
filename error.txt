# Define the path to the kubeconfig file
$kubeconfigPath = "$env:USERPROFILE\.kube\config"

# Check if the kubeconfig file contains "***"
if (Test-Path $kubeconfigPath) {
    # Search for the string "***" in the file
    $containsMaskedData = Select-String -Path $kubeconfigPath -Pattern "\*\*\*" -Quiet

    if ($containsMaskedData) {
        Write-Host "The kubeconfig file contains masked data (***)."
    } else {
        Write-Host "The kubeconfig file does not contain masked data (***)."
    }
} else {
    Write-Host "The kubeconfig file does not exist at: $kubeconfigPath"
}
